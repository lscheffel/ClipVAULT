    <script>
        async function loadHistory() {
            const result = await window.pywebview.api.get_history();
            const list = document.getElementById('history-list');
            list.innerHTML = '';

            result.forEach(entry => {
                const div = document.createElement('div');
                div.className = 'entry';

                const time = document.createElement('time');
                time.textContent = new Date(entry.timestamp).toLocaleString();
                div.appendChild(time);

                const content = document.createElement('div');
                content.className = 'entry-content';

                if (entry.type === 'text') {
                    content.textContent = entry.content;
                } else if (entry.type === 'image') {
                    const img = document.createElement('img');
                    img.src = entry.path || entry.content;
                    img.className = 'media-preview';
                    content.appendChild(img);
                } else if (entry.type === 'video') {
                    const video = document.createElement('video');
                    video.src = entry.path || entry.content;
                    video.controls = true;
                    video.className = 'media-preview';
                    content.appendChild(video);
                } else if (entry.type === 'audio') {
                    const audio = document.createElement('audio');
                    audio.src = entry.path || entry.content;
                    audio.controls = true;
                    audio.className = 'media-preview';
                    content.appendChild(audio);
                } else {
                    content.textContent = `[${entry.type}] ${entry.content}`;
                }

                div.appendChild(content);

                const actions = document.createElement('div');
                actions.className = 'entry-actions';

                const saveBtn = document.createElement('button');
                saveBtn.textContent = 'Salvar como...';
                saveBtn.className = 'save';
                saveBtn.onclick = () => window.pywebview.api.save_entry(entry.id);
                actions.appendChild(saveBtn);

                const delBtn = document.createElement('button');
                delBtn.textContent = 'Excluir';
                delBtn.className = 'delete';
                delBtn.onclick = () => {
                    if (confirm('Deseja realmente excluir este item?')) {
                        window.pywebview.api.delete_entry(entry.id);
                        loadHistory(); // atualiza a lista
                    }
                };
                actions.appendChild(delBtn);

                div.appendChild(actions);
                list.appendChild(div);
            });
        }

        window.onload = loadHistory;
    </script>
